Function GetWishlist:
    If userID is not found:
        Return HTTP 401 Unauthorized with message "User not authenticated"

# Remove product from wishlist
Function RemoveFromWishlist(productID):
    If userID is not found:
        Return HTTP 401 Unauthorized with message "User not authenticated"

    result = deleteWishlistItem(userID, productID)
    If result is error:
        Return HTTP 500 Internal Server Error with message "Failed to remove item from wishlist"

    Return HTTP 200 OK with message "Item removed from wishlist"






# Delete all items from wishlist
Function ClearWishlist:
    If userID is not found:
        Return HTTP 401 Unauthorized with message "User not authenticated"

    result = clearWishlistByUserID(userID)
    If result is error:
        Return HTTP 500 Internal Server Error with message "Failed to clear wishlist"

    Return HTTP 200 OK with message "Wishlist cleared"
